    SQLC: The Anti-ORM

    Cut out the middle-man, and save!

    ./sql/queries/todos.sql:

    -- name: All :many
SELECT
    "uuid",
    "task",
    "priority",
    "group",
    "created_at",
    "completed_at"
FROM
    "todos"
WHERE
    "group" = ANY(sqlc.arg('group')::text[])
ORDER BY
    "priority"
;

    SQLC Annotations

    -- name: <name> <command>

    -- name: InsertOne :exec
-- name: InsertAndReturnOne :one

-- name: ByID :one
-- name: ByGroup :many

    See https://docs.sqlc.dev/en/latest/reference/query-annotations.html.

    Result Annotations

    -- name: InsertOne :execlastid
-- (returns id)

-- name: DeleteSome :execresult
-- (returns number of affected rows)

-- name: DeleteAll :execrows
-- (returns affected rows)

    See https://docs.sqlc.dev/en/latest/reference/query-annotations.html.

    sqlc.argv('priority'),

    sqlc.argv('completed_at')

    version: "2"
sql:
  - engine: "postgresql"
    queries: "./sql/queries/todos/"
    schema: "./sql/migrations/"
    gen:
      go:
        sql_package: "pgx/v5"
        package: "todos"
        out: "./db/todos/"
        emit_json_tags: true
        json_tags_case_style: "snake"
        overrides:
          - column: "todos.created_at"
            go_struct_tag: 'json:"-"'
